
#include <stdio.h>
#include<stdlib.h>

int main()
{
    
    char ch;
    int temp[100],i,n,flag1,flag2,flag3,s;
    goto menu;
    
    menu:
   printf("\t\t\t\t======= BD Meteorological System =========\n");
   printf("\t\t\t\t\t\tMENU\n\n");
   printf("\t\t\tEnter ‘r’, to take temperatures of all cities\n");
   printf("\t\t\tEnter ‘s’, to search a temperature\n");
   printf("\t\t\tEnter ‘t’, to display all the temperatures in reverse\n");
   printf("\t\t\tEnter ‘q’, to quit/exit the menu system\n");
   take:
    printf("Enter the Number of all cities:\n");
    scanf("%d",&n);
    printf("Enter the Tempratures of all cities:\n");
    for(i=0; i<n; i++){
        scanf("%d",&temp[i]);
    }
   
    quit:
    exit(1);
    reverse:
    for(i=0; i<n; i++){
        
        if(temp[i]==NULL){
            flag3=1;
            break;
            
        }
        else{
         flag3=0;
        }
      
    }
    if(flag3==1){
        printf("No temperature information is found. Please populate your temperature information first");
    }
    else{
    printf("The Tempratures of all cities in reverse order:");
    for(i=n-1; i<=0; i--){
        
        printf("%d",temp[i]);
        
    }
    }
    search:
     for(i=0; i<n; i++){
        
        if(temp[i]==NULL){
            flag1=1;
            break;
            
        }
        else{
         flag1=0;
        }
        
    }
    if(flag1==1){
        printf("No temperature information is found. Please populate your temperature information first");
    }
    else{
    
       printf("Enter the temprature you want to search:"); 
       scanf("%d",&s);
       
    for(i=0; i<n; i++){
       if(temp[i]==s){
         flag2=1;
         break;
       }
    else{
        flag2=0;
    }
       
    }
    if(flag2==1){
         
          printf("The Index No. is:%d",i);
    }
    else{
        printf("Not Found");
    }
   
    
   printf("Enter Your Choice:\n");
   scanf("%c ",&ch);
   switch(ch){
       case 'r':
       goto take;
       break;
       case's':
       goto search;
       break;
       case't':
       goto reverse;
       goto menu;
       break;
       case 'q':
       goto quit;
       break;
       
       default:
       printf("Invalid input");
   }

    return 0;
}
