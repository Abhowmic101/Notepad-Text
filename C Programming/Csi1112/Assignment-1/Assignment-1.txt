 #include <stdio.h>

int main()
{
    int map[50],i,n,max,min,sub=0;
    scanf("%d",&n);
    for(i=0; i<n; i++){
        
        scanf("%d",&map[i]);
        
    }
    max=map[0];
    min=map[0];
    for(i=0; i<n; i++){
    
        if(max<=map[i]){
            max=map[i];
    }
    
    }
    
    
    for(i=0; i<n; i++){
    
        if(min>map[i]){
            min=map[i];
    }
  
    }
    sub=max-min;
    printf("Maximum Profit:%d",sub);
      
      
      
      
    return 0;
}
